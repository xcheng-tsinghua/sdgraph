
## 同步到云端
# 查看当前状态
git status

# 将修改添加到暂存区
git add .

# 将更改信息暂存到本地
git commit -m "change"

# 推送到远程仓库
git push origin main





## 同步到本地
git pull origin main

# 强制同步到本地
git reset --hard
git pull origin main




## 查看 nohup 的进程输出
tail -f nohup.out



## 服务器训练
nohup python train_cls.py --bs=32 --local=False
